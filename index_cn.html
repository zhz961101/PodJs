双向绑定

let app = Poi({
    el: "#app1",
    data: {
        msg: "Poi! "
    }
})
    Html
    
<p>{{ msg }}</p>
<input type="text" name="" bind:value="msg" on:keyup="msg=self.value">

数据驱动

                    JS
                    
let app = Poi({
    el: "#app2",
    data: {
        msg: "Poi! ",
        reverseMsg: function() {
            this.msg = this.msg.split('').reverse().join('')
        }
    }
})
    Html
    
<input type="text" name="" bind:value="msg" on:keyup="msg=self.value">
<button type="button" name="button" on:click="reverseMsg()">reverse text</button>
<button type="button" name="button" on:click="msg=msg+msg">deouble</button>
<p>{{ msg }}</p>

模板引擎 - 支持原生Js

                    JS
                    
let app = Poi({
    el: "#app3",
    data: {
        items: ["push random button"],
        addRandom: function() {
            this.items.push(Math.floor((Math.random() * 100)))
        }
    }
})
    Html
    
<button type="button" name="button" on:click="addRandom()">
    push random item!<button>
<ul>
    {{ for(let item of items){ }}
        <li>{{ item }}</li>
    {{ } }}
</ul>
